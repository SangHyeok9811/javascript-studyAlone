Array - 데이터의 나열
        
        선언
        var 배열명 - 변수규칙

        var arr = new Array()
        arr[0] = 데이터1
        arr[1] = 데이터2
        arr[2] = 데이터3
        arr[3] = 데이터4

        var arr = [];
        arr[0] = 데이터1;
        arr[1] = 데이터2;
        arr[2] = 데이터3;
        arr[3] = 데이터4;

        var arr = new Array(데이터1, 데이터2, 데이터3, 데이터4 ...)
        var arr = [데이터1, 데이터2, 데이터3, 데이터4 ...]
